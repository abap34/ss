Document block, control flow, and properties
================================================================================

document
  page_no_all()
end

page "Control"
  let title = text("Control")
  let items = list("a", "b")
  for item, index in items
    if index then
      property title "data-kind" item
    else
      set_prop(title, "fallback", c"#fff")
    end
  end
end
