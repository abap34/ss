const plus_one: number -> number = (x: number) |-> x + 1

fn make_label(text_value: string) -> page -> object
  return (page_value: page)
    |->
      place_on!(page_value, new(text_value, "body", "text"))
end

fn use(callback: (page -> object) -> document, pair: (page, document) -> object, thunk: () -> number) -> void
  foreach(pages(docctx()), (page_value: page) |-> pair(page_value, docctx()))
  return
end

fn/! note(text_value: string) -> object
  return text(text_value)
end

fn place_direct!(obj: object) -> object
  return place!(obj)
end
